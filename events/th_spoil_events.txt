add_namespace = spoil_event
country_event = {
	#xuanwuchuan
	id = spoil_event.1
	title = spoil_event.1.t
	desc = spoil_event.1.d
	fire_only_once = no
	is_triggered_only = no
	mean_time_to_happen = {
		days = 2
	}
	trigger = {
		69 = {
			is_controlled_by = ROOT
		}
		69 = {
			NOT = {
				is_owned_by = ROOT
			}
		}
		ROOT = {
			has_country_flag = Spoil_xiaxuanwucheng
		}
	}
	option = {
		#革命
		name = spoil_event.1.1
		clr_country_flag = Spoil_xiaxuanwucheng
		if
		{
			limit = {
				has_idea = SGS_mic_spoil0_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil0_idea
				add_idea = SGS_mic_spoil5_idea
			}
		}

		
		every_country = {
			limit = {
					owns_state = 69
			}
			white_peace = ROOT
			country_event = { id = spoil_event.100 days = 1 random = 1 }
			
		}
		ai_chance = {
			base = 35
		}
	}
}
